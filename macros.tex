%------------------------------------------------------------------------------
% Macros.
%------------------------------------------------------------------------------

% Real numbers.
\newcommand{\R}[0]{\mathds{R}}

% Integers.
\newcommand{\Z}[0]{\mathds{Z}}

% Natural numbers.
\newcommand{\N}[0]{\mathds{N}}

% Dataset D.
\newcommand{\D}[0]{\mathcal{D}}

% Expectation.
\newcommand{\E}[0]{\mathds{E}}

% Vector.
\renewcommand{\vec}[1]{{\boldsymbol{{#1}}}}

% Matrix.
\newcommand{\mat}[1]{{\boldsymbol{{#1}}}}

% Transpose.
\newcommand{\T}[0]{^\top}

% Inverse.
\newcommand{\inv}{^{-1}}

% Norm.
\newcommand{\norm}[1]{\left\lVert#1\right\rVert}

% Trace.
\newcommand{\tr}[1]{\mathrm{tr}(#1)}

% Determinant.
\newcommand{\determ}[1]{\mathrm{det}(#1)}

% Theta minimizer.
\newcommand{\thetaopt}[0]{\vec{\theta^*}}

% Gaussian N(#1, #2).
\newcommand{\gauss}[2]{\mathcal{N}\big(#1,\,#2\big)}
\newcommand{\gaussBig}[2]{\mathcal{N}\left(#1,\,#2\right)}

% Gaussian N(#1 | #2, #3).
\newcommand{\gaussx}[3]{\mathcal{N}\big(#1\,|\,#2,\,#3\big)}
\newcommand{\gaussxBig}[3]{\mathcal{N}\left(#1\,|\,#2,\,#3\right)}

% Covariance matrix.
\DeclareMathOperator{\cov}{Cov}
